Rendering API

        
// Render to canvas
UPCARenderer.render('#my-canvas', '012345678905');

// With options
UPCARenderer.render(canvas, '012345678905', {
    moduleWidth: 2,          // Bar width (px)
    height: 70,              // Main bar height (px)
    guardExtend: 10,         // Guard bar extension (px)
    fontSize: 14,            // Middle digit font size (px)
    smallDigitFontSize: 10,  // First/last digit font size (px)
    textMargin: 2,           // Gap between bars and text
    quietZone: 9,            // Quiet zone (in modules)
    style: 'notched',        // 'notched' | 'flat'
    checksum: 'auto',        // 'auto' | 'validate' | 'recalculate'
    background: '#FFFFFF',
    foreground: '#000000'
});

// Get as Data URL
const dataUrl = UPCARenderer.toDataURL('012345678905');

// Get as Blob (async)
const blob = await UPCARenderer.toBlob('012345678905');

// Get as SVG string
const svgString = UPCARenderer.toSVG('012345678905');

    


    

        
Input Flexibility

        
// 11 digits - checksum auto-calculated
UPCARenderer.render(canvas, '01234567890');  // → 012345678905

// 12 digits - checksum validated/used
UPCARenderer.render(canvas, '012345678905');

// Shorter codes - padded with leading zeros
UPCARenderer.render(canvas, '12345');  // → 000001234506

// Checksum modes
UPCARenderer.render(canvas, '012345678905', { checksum: 'auto' });
// 'auto' - accept if valid, recalculate if invalid

UPCARenderer.render(canvas, '012345678905', { checksum: 'validate' });
// 'validate' - throw error if checksum is wrong

UPCARenderer.render(canvas, '012345678900', { checksum: 'recalculate' });
// 'recalculate' - always replace last digit with correct checksum

    


    

        
Validation API

        
// Validate a UPC-A code (checks checksum for 12-digit input)
UPCARenderer.validate('012345678905');  // true
UPCARenderer.validate('012345678900');  // false (wrong checksum)

// Calculate the check digit for an 11-digit code
UPCARenderer.calculateChecksum('01234567890');  // 5

// Format UPC-A with hyphens
UPCARenderer.formatUPC('012345678905');  // "0-12345-67890-5"

// Get full encoding with auto-calculated checksum
UPCARenderer.encode('01234567890');
// { encoding: '10100...', fullCode: '012345678905' }
